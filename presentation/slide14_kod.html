domain/ApprovalPolicy.py — deterministyczna reguła progu 1000 PLN
class ApprovalPolicy:
    def __init__(self):
        self.log = Logger.get()
        self.config = {
            "THRESHOLD_PLN": Decimal(os.getenv("APPROVAL_THRESHOLD_PLN", "1000")),
        }

    def evaluate(self, amount: Decimal, currency="PLN") -> ApprovalRoute:
        threshold = self.config["THRESHOLD_PLN"]
        if amount > threshold:                       # > 1000 PLN
            return ApprovalRoute(ApprovalLevel.DIRECTOR, False, "Dyrektor")
        return ApprovalRoute(ApprovalLevel.AUTO, True, "auto/przełożony")

class RequestService:
    def submit(self, request: PurchaseRequest) -> PurchaseRequest:
        route = self.policy.evaluate(request.amount, request.currency)
        request.request_id = self.repo.save(request)         # Azure SQL (3NF)
        self.archive.append("RequestSubmitted", request)      # Blob WORM
        if route.auto_approve:
            return self._finalize(request, RequestStatus.AUTO_APPROVED)
        self.notifier.publish("ApprovalRequired", {...})       # Service Bus
        return request
