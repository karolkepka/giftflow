schema.sql — tabela 3NF + reguła progu (CHECK / kolumna wyliczana)
CREATE TABLE dbo.purchase_request (
  request_id     INT IDENTITY PRIMARY KEY,
  employee_id    INT NOT NULL REFERENCES dbo.employee(employee_id),
  gift_type_id   INT NOT NULL REFERENCES dbo.gift_type(gift_type_id),
  amount         DECIMAL(12,2) NOT NULL CHECK (amount > 0),
  currency       CHAR(3) NOT NULL DEFAULT 'PLN',
  status         VARCHAR(16) NOT NULL DEFAULT 'PENDING'
    CHECK (status IN ('PENDING','AUTO_APPROVED','APPROVED','REJECTED')),
  created_at     DATETIME2 NOT NULL DEFAULT SYSUTCDATETIME()
);
-- próg 1000 PLN wymuszony też w danych (defense in depth)
ALTER TABLE dbo.purchase_request ADD requires_director AS

fact_spend — widok analityczny (star schema)
CREATE VIEW fact_spend AS
SELECT r.request_id, e.department_id, g.category,
       CONVERT(date, a.decided_at) AS decision_date, r.amount
FROM dbo.purchase_request r
JOIN dbo.approval  a ON a.request_id = r.request_id AND a.decision='APPROVE'
JOIN dbo.employee  e ON e.employee_id = r.employee_id
JOIN dbo.gift_type g ON g.gift_type_id = r.gift_type_id;
